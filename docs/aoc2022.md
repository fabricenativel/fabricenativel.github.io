```python
reader = open("inputday8.txt")
forest = reader.read().strip().split('\n')

def visible(line,number,is_row):
    lindex = {0,len(line)-1} | {i for i in range(1,len(line)-1) if line[i]>max(line[0:i]) or line[i]>max(line[i+1:])}
    return {(number,i) for i in lindex} if is_row else {(i,number) for i in lindex}

visible_tree = set()
for l in range(len(forest)):
    visible_tree.update(visible(forest[l],l,True))
    column = [forest[k][l] for k in range(len(forest))]
    visible_tree.update(visible(column,l,False))

print(len(visible_tree))
```

```python
reader = open("inputday8.txt")
forest = reader.read().strip().split('\n')
directions = [(1,0),(-1,0),(0,1),(0,-1)]
size  =  len(forest)

def in_forest(l,c):
    return 0<=l<size and 0<=c<size

def scenic_score(l,c):
    sc = 1
    for dl,dc in directions:
        k = 1
        while in_forest(l+k*dl,c+k*dc) and int(forest[l][c])>int(forest[l+k*dl][c+k*dc]):
            k += 1
        if not in_forest(l+k*dl,c+k*dc): k -= 1
        sc = sc*k
    return sc

print(max(scenic_score(l,c) for l in range(1,size-1) for c in range(1,size-1)))
```

```python
shortest = {(i,j):inf for i in range(height) for j in range(width)}
to_look = queue.deque()

to_look.append(start)
while len(to_look)!=0:
    current = to_look.pop()
```

```python
import queue
from math import inf

DIRECTIONS = [(1,0),(-1,0),(0,1),(0,-1)]

reader = open("inputday12.txt")
map  = reader.read().strip().split('\n')
width,height = (len(map[0]),len(map))
starting_points = []
for i in range(len(map)):
    if 'E' in map[i]:
        exit = (i,map[i].index('E'))
    if 'S' in map[i]:
        start = (i,map[i].index('S'))
        starting_points.append(start)
    if 'a' in map[i]:
        for j in range(width):
            if map[i][j]=='a':
                starting_points.append((i,j))

map[start[0]]=map[start[0]].replace('S','a')



def reachable(l,c):
    return [(l+i,c+j) for i,j in DIRECTIONS if 0<=l+i<height and 0<=c+j<width and ord(map[l+i][c+j])<=ord(map[l][c])+1]

min_trail = inf
for start in starting_points:
    shortest = {(i,j):inf for i in range(height) for j in range(width)}
    to_look = queue.deque()
    shortest[start]=0
    to_look.append(start)
    while len(to_look)!=0:
        current = to_look.pop()
        for neighbour in reachable(*current):
            if shortest[neighbour]>shortest[current]+1:
                shortest[neighbour] = shortest[current]+1
                to_look.appendleft(neighbour)
    if shortest[exit]<min_trail:
        min_trail = shortest[exit]

print(min_trail)
```
